Cada sitio de Mintlify necesita un archivo `docs.json` con la configuración básica. Aprende más sobre las [propiedades](#propiedades) a continuación.

## Propiedades

<ResponseField name="name" type="string" required>
  Nombre de tu proyecto. Usado para el título global.

  Ejemplo: `mintlify`
</ResponseField>

<ResponseField name="navigation" type="Navigation[]" required>
  Un array de grupos con todas las páginas dentro de ese grupo

  <Expandable title="Navegación">
    <ResponseField name="group" type="string">
      El nombre del grupo.

      Ejemplo: `Configuración`
    </ResponseField>
    <ResponseField name="pages" type="string[]">
      Las rutas relativas a los archivos markdown que servirán como páginas.

      Ejemplo: `["personalizacion", "pagina"]`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string or object">
  Ruta a la imagen del logo u objeto con ruta al logo en modo "claro" y "oscuro"

  <Expandable title="Logo">
    <ResponseField name="light" type="string">
      Ruta al logo en modo claro
    </ResponseField>
    <ResponseField name="dark" type="string">
      Ruta al logo en modo oscuro
    </ResponseField>
    <ResponseField name="href" type="string" default="/">
      A dónde te lleva el clic en el logo
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  Ruta a la imagen del favicon
</ResponseField>

<ResponseField name="colors" type="Colors">
  Códigos de color hexadecimal para tu tema global

  <Expandable title="Colores">
    <ResponseField name="primary" type="string" required>
      El color primario. Usado más frecuentemente para contenido resaltado, encabezados de sección, acentos, en modo claro
    </ResponseField>
    <ResponseField name="light" type="string">
      El color primario para modo oscuro. Usado más frecuentemente para contenido resaltado, encabezados de sección, acentos, en modo oscuro
    </ResponseField>
    <ResponseField name="dark" type="string">
      El color primario para botones importantes
    </ResponseField>
    <ResponseField name="background" type="object">
      El color del fondo tanto en modo claro como oscuro

      <Expandable title="Objeto">
        <ResponseField name="light" type="string" required>
          El código de color hexadecimal del fondo en modo claro
        </ResponseField>
        <ResponseField name="dark" type="string" required>
          El código de color hexadecimal del fondo en modo oscuro
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  Array de `nombre`s y `url`s de enlaces que quieres incluir en la barra superior

  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">
      El nombre del botón.

      Ejemplo: `Contáctanos`
    </ResponseField>
    <ResponseField name="url" type="string">
      La url cuando haces clic en el botón. Ejemplo: `https://mintlify.com/docs`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  <Expandable title="Llamada a la Acción de la Barra Superior">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
      Link muestra un botón. GitHub muestra la información del repositorio en la url proporcionada incluyendo el número de estrellas de GitHub.
    </ResponseField>
    <ResponseField name="url" type="string">
      Si es `link`: A qué enlaza el botón.

      Si es `github`: Enlace al repositorio del que cargar la información de GitHub.
    </ResponseField>
    <ResponseField name="name" type="string">
      Texto dentro del botón. Solo requerido si `type` es `link`.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="versions" type="string[]">
  Array de nombres de versiones. Solo usa esto si quieres mostrar diferentes versiones
  de documentación con un desplegable en la barra de navegación.
</ResponseField>

<ResponseField name="anchors" type="Anchor[]">
  Un array de los anclajes, incluye el `icon`, `color`, y `url`.

  <Expandable title="Anclaje">
    <ResponseField name="icon" type="string">
      El icono de [Font Awesome](https://fontawesome.com/search?q=heart) usado para destacar el anclaje.

      Ejemplo: `comments`
    </ResponseField>
    <ResponseField name="name" type="string">
      El nombre de la etiqueta del anclaje.

      Ejemplo: `Comunidad`
    </ResponseField>
    <ResponseField name="url" type="string">
      El inicio de la URL que marca qué páginas van en el anclaje. Generalmente, este es el nombre de la carpeta donde pones tus páginas.
    </ResponseField>
    <ResponseField name="color" type="string">
      El color hexadecimal del fondo del icono del anclaje. También puede ser un gradiente si pasas un objeto con las propiedades `from` y `to` que son cada una un color hexadecimal.
    </ResponseField>
    <ResponseField name="version" type="string">
      Usado si quieres ocultar un anclaje hasta que se seleccione la versión correcta de la documentación.
    </ResponseField>
    <ResponseField name="isDefaultHidden" type="boolean" default="false">
      Pasa `true` si quieres ocultar el anclaje hasta que enlaces directamente a alguien a la documentación dentro de él.
    </ResponseField>
    <ResponseField name="iconType" type="string" default="duotone">
      Uno de: "brands", "duotone", "light", "sharp-solid", "solid", o "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  Anula las configuraciones predeterminadas para el anclaje superior.

  <Expandable title="Objeto">
    <ResponseField name="name" type="string" default="Documentation">
      El nombre del anclaje superior
    </ResponseField>
    <ResponseField name="icon" type="string" default="book-open">
      Icono de Font Awesome.
    </ResponseField>
    <ResponseField name="iconType" type="string" default="duotone">
      Uno de: "brands", "duotone", "light", "sharp-solid", "solid", o "thin"
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  Un array de pestañas de navegación.

  <Expandable title="Pestañas">
    <ResponseField name="name" type="string">
      El nombre de la etiqueta de la pestaña.
    </ResponseField>
    <ResponseField name="url" type="string">
      El inicio de la URL que marca qué páginas van en la pestaña. Generalmente, este
      es el nombre de la carpeta donde pones tus páginas.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="api" type="API">
  Configuración para ajustes de API. Aprende más sobre las páginas de API en [Componentes de API](/api-playground/demo).

  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">
      La url base para todos los endpoints de API. Si `baseUrl` es un array, habilitará múltiples opciones
      de url base que el usuario puede alternar.
    </ResponseField>
    <ResponseField name="auth" type="Auth">
      <Expandable title="Auth">
        <ResponseField name="method" type='"bearer" | "basic" | "key"'>
          La estrategia de autenticación usada para todos los endpoints de API.
        </ResponseField>
        <ResponseField name="name" type="string">
          El nombre del parámetro de autenticación usado en el playground de API.

          Si el método es `basic`, el formato debe ser `[nombreUsuario]:[nombreContraseña]`
        </ResponseField>
        <ResponseField name="inputPrefix" type="string">
          El valor predeterminado que está diseñado para ser un prefijo para el campo de entrada de autenticación.

          Por ejemplo, si un `inputPrefix` de `ClaveAuth` heredaría el resultado de entrada predeterminado del campo de autenticación como `ClaveAuth`.
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="playground" type="Playground">
      Configuraciones para el playground de API

      <Expandable title="Playground">
        <ResponseField name="mode" type='"show" | "simple" | "hide"' default="show">
          Si el playground se está mostrando, oculto, o solo mostrando el endpoint sin interactividad del usuario agregada `simple`

          Aprende más en las [guías del playground](/api-playground/demo)
        </ResponseField>
      </Expandable>
    </ResponseField>
    <ResponseField name="maintainOrder" type="boolean">
      Habilitar esta bandera asegura que el orden de las claves en las páginas de OpenAPI coincida con el orden de las claves definido en el archivo OpenAPI.

      <Warning>
        Este comportamiento pronto estará habilitado por defecto, momento en el cual este campo será obsoleto.
      </Warning>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  Una cadena o un array de cadenas de URL(s) o ruta(s) relativa(s) que apuntan a tu
  archivo OpenAPI.

  Ejemplos:

  <CodeGroup>

  ```json Absoluta
  "openapi": "https://ejemplo.com/openapi.json"
  ```

  
  ```json Relativa
  "openapi": "/openapi.json"
  ```

  
  ```json Múltiple
  "openapi": ["https://ejemplo.com/openapi1.json", "/openapi2.json", "/openapi3.json"]
  ```

  </CodeGroup>
</ResponseField>

<ResponseField name="footerSocials" type="FooterSocials">
  An object of social media accounts where the key:property pair represents the social media platform and the account url.

  Example:

  ```json
  {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  }
  ```

  <Expandable title="FooterSocials">
    <ResponseField name="[key]" type="string">
      One of the following values `website`, `facebook`, `x`, `discord`, `slack`, `github`, `linkedin`, `instagram`, `hacker-news`

      Example: `x`
    </ResponseField>
    <ResponseField name="property" type="string">
      The URL to the social platform.

      Example: `https://x.com/mintlify`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="feedback" type="Feedback">
  Configurations to enable feedback buttons

  <Expandable title="Feedback">
    <ResponseField name="suggestEdit" type="boolean" default="false">
      Enables a button to allow users to suggest edits via pull requests
    </ResponseField>
    <ResponseField name="raiseIssue" type="boolean" default="false">
      Enables a button to allow users to raise an issue about the documentation
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="modeToggle" type="ModeToggle">
  Customize the dark mode toggle.

  <Expandable title="ModeToggle">
    <ResponseField name="default" type={'"light" or "dark"'}>
      Set if you always want to show light or dark mode for new users. When not
      set, we default to the same mode as the user's operating system.
    </ResponseField>
    <ResponseField name="isHidden" type="boolean" default="false">
      Set to true to hide the dark/light mode toggle. You can combine `isHidden` with `default` to force your docs to only use light or dark mode. For example:

      <CodeGroup>

      ```json Only Dark Mode
      "modeToggle": {
        "default": "dark",
        "isHidden": true
      }
      ```

      
      ```json Only Light Mode
      "modeToggle": {
        "default": "light",
        "isHidden": true
      }
      ```

      </CodeGroup>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  A background image to be displayed behind every page. See example with
  [Infisical](https://infisical.com/docs) and [FRPC](https://frpc.io).
</ResponseField>
